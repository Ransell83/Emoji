var RightEyeColourDot = $("#right-eye-dot");
var LeftEyeColourDot = $("#left-eye-dot");

RightEyeColourDot.click(function() {
	RightEyeColourDot.attr("fill", "red");
});

LeftEyeColourDot.click(function() {
	LeftEyeColourDot.attr("fill", "blue");
});

var faceShape = $("#face");

var FaceOrange;
FaceOrange = true;

var changeMouth = $("#mouth");

var Puzzled;
Puzzled = true;

changeMouth.click(function() {

    if(Puzzled == true){
        // Face is smiley - make it puzzled
        changeMouth.attr("d", "M 250,450 q 100,-200 100,0 z");

        Puzzled = false;

    } else {
        
        changeMouth.attr("d", "M 150,360 q 150,200 300,0 z");
        
        Puzzled = true;
    }
    
    if(FaceOrange == true){
        // Face is orange - make it yellow
        faceShape.attr("fill", "yellow");

        FaceOrange = false;

    }
});
